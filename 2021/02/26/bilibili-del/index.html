自动删除B站抽奖动态

2021-02-26- 2021-12-17
233- 1m
黑科技

在登录状态下打开我的动态页面

下面用我的小号用来展示

复制下面的代码

let delay = 1000,        // 删除时间间隔
    scrolls = 800;       // 滚动距离
let info = '',           // 信息
    rolls = 0;           // 距离
//询问是否删除
if (/dynamic/.test(window.location.href) && confirm("确定删除抽奖吗")) {
    Fun();
}
function Fun() {
    rolls += scrolls;
    $('html, body').animate({scrollTop: rolls}, 30);
    // 展开内容
    $(".fold-text").click();
    $(".expand-btn").click();
    // 定位抽奖内容
    info = $(".main-content").find('span[click-title="抽奖详情"]') || $(".main-content").find('.deleted-text');
    setTimeout(getLottery, delay);
}
//获取
function getLottery() {
    info = info.parents(".card");
    info[info.length - 1].querySelectorAll(".child-button")[1].click();
    setTimeout(clickDel, delay);
}
//删除
function clickDel() {
    $(".popup-content-ctnr")[$(".popup-content-ctnr").length - 2].querySelector(".bl-button").click();
    Fun()
}

在键盘上按下F12，呼出控制台(console)，按顺序操作

第三步如上图，在空白处粘贴代码按回车即可
